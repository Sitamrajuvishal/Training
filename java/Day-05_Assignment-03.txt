Q. Implement a Matrix class that has a constructor which initializes the dimensions of a matrix and a method to fill the matrix with values.

public class Matrix {
    private int rows;
    private int cols;
    private int[][] data;

    public Matrix(int rows, int cols) {
        this.rows = rows;
        this.cols = cols;
// Initialize matrix with all zeros
        this.data = new int[rows][cols]; 
    }

    public void fill(int[] values) {
        if (values.length != rows * cols) {
            throw new IllegalArgumentException("Number of values doesn't match matrix dimensions");
        }

        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                data[i][j] = values[i * cols + j];
            }
        }
    }
